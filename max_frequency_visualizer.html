def maxFrequency(arr, k):
    arr.sort()                         # sort ascending
    left, window_sum = 0, 0
    best = 1

    for right in range(len(arr)):
        window_sum += arr[right]           # expand right
        window_size = right - left + 1
        cost = arr[right] * window_size - window_sum

        while cost > k:                   # shrink if too costly
            window_sum -= arr[left]
            left += 1
            window_size -= 1
            cost = arr[right] * window_size - window_sum

        best = max(best, right - left + 1)

    return best
